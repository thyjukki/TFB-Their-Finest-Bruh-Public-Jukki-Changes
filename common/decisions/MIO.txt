		icon = generic_propaganda

		}

		priority = 10
        icon = GFX_decision_generic_tank
		allowed = {
			always = yes
		}
		visible = {
			is_ai = no
            NOT = { has_variable = donor_MIO }
            has_variable = tank_mio_displayed
			FROM = {
				var:target_MIO@ROOT = {
                    is_mio_available = yes
					has_mio_flag = tank_mio
                    NOT = { has_mio_flag = has_transfered }
                    has_mio_size > 2
				}
			}
		}

		on_map_mode = decision_view_only

		state_target = yes
		target_array = ROOT.target_MIO_state

		target_trigger = {

		}
		
		fire_only_once = no

		complete_effect = {
            set_variable = { donor_MIO = FROM.target_MIO@ROOT }
            var:donor_MIO = { set_mio_flag = active_source }
            add_to_variable = { MIO_Transfer_GUI_dirty_var = 1 }	
		}
	}
